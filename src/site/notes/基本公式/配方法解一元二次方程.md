$$
ax^2+bx+c=0
$$
$$
x^2+\frac{b}{a}x=-\frac ca
$$
$$
x^2+\frac{b}{a}x+(\frac{b}{2a})^2=-\frac ca+(\frac{b}{2a})^2
$$
$$
(x+\frac{b}{2a})^2=-\frac ca+(\frac{b}{2a})^2
$$
$$
x+\frac{b}{2a}=\pm\sqrt{-\frac ca+(\frac{b}{2a})^2}
$$
